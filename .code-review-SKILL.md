```python
def conjugate_gradient(A: Matrix, b: Vector, x0: Vector | None = None, 
                       tol: Scalar = 1e-6) -> tuple[Vector, int]:

    前提条件：A は対称正定値であること。
```

**テスト テンプレート**:
```python
def test_conjugate_gradient_convergence_with_spd_matrix():
    """共役勾配法が対称正定値行列で収束することを検証。"""
    # Arrange: テストデータ設定
    n = 10
    A = jnp.eye(n) * 2.0  # 対称正定値行列
    b = jnp.ones(n)
    x_expected = b / 2.0  # 解は b/2
    
    # Act: テスト対象関数実行
    x, iterations = conjugate_gradient(A, b, tol=1e-10)
    
    # Assert: 検証
    assert iterations < 5 * n, f"収束が遅い: {iterations} 回"
    assert jnp.allclose(x, x_expected, atol=1e-8)
```

#### テストカバレッジ
- [ ] 正常系（happy path）でテストがあるか
- [ ] 境界ケース（empty、単一要素、最大値）がカバーされているか

**必須テストケース例**:
```python
def test_solver_with_empty_input():
    """空入力を拒否することを検証。"""
    with pytest.raises(ValueError, match="empty"):
        solver(A=jnp.array([]).reshape(0, 0), b=jnp.array([]))

def test_solver_with_single_element():
    """スカラー（1x1 行列）で正しく動作することを検証。"""
    A = jnp.array([[2.0]])
    b = jnp.array([4.0])
    x, _ = solver(A, b)
    assert jnp.allclose(x, jnp.array([2.0]))

def test_solver_with_nan_input():
    """NaN 入力を適切に処理することを検証。"""
    A = jnp.array([[jnp.nan, 0.0], [0.0, 2.0]])
    b = jnp.array([1.0, 2.0])
    with pytest.raises((ValueError, RuntimeError)):
        solver(A, b)
```

#### テスト可読性
- [ ] パラメータ化テスト（`@pytest.mark.parametrize`）が適切に使用されているか
- [ ] fixture の目的が明確か
- [ ] assertion メッセージが情報的か（失敗時に原因が分かるか）


**整合性チェック例**:
```python
def conjugate_gradient(A: Matrix, b: Vector, tol: Scalar = 1e-6) -> tuple[Vector, int]:
    """共役勾配法による線形方程式求解。
    
    前提条件：A は対称正定値であること。    ← テストで検証必須
    
    Returns:
        (x, iterations): 解 x と反復回数    ← テストで両方を assert 必須
        
    Raises:
        ValueError: A が正方行列でない場合  ← テストで ValueError をキャッチ必須
        RuntimeError: A が対称正定値でない場合
    """
    if A.shape[0] != A.shape[1]:
        raise ValueError("A must be square")
    # ...
    return x, iterations

# 対応するテスト
def test_conjugate_gradient_raises_for_non_square():
    """A が非正方行列の場合 ValueError を厳発すること。"""
    A = jnp.ones((3, 4))
    b = jnp.ones(3)
    with pytest.raises(ValueError, match="square"):
        conjugate_gradient(A, b)  # ← Raises セクションをカバー

def test_conjugate_gradient_returns_tuple():
    """戻り値が (x, iterations) タプルであることを検証。"""
    A = jnp.eye(5) * 2.0
    b = jnp.ones(5)
    result = conjugate_gradient(A, b)
    assert isinstance(result, tuple)
    assert len(result) == 2
    x, iterations = result
    assert isinstance(x, jnp.ndarray)
    assert isinstance(iterations, (int, jnp.integer))
```

#### 型安全性
- [ ] implicit type conversion が避けられているか
- [ ] null pointer が適切にチェックされているか
- [ ] 整数オーバーフローリスクはないか
- [ ] `const` / `constexpr` が適切に付与されているか

**チェックコード例**:
```cpp
// ❌ NG: implicit conversion + null pointer チェックなし
void process_matrix(const double* data, int size) {
    int* buffer = data;  // double* -> int* (危険な implicit conversion)
    for (int i = 0; i < size; ++i) {
        buffer[i] = (int)data[i];  // オーバーフロー risk なし
    }
}

// ✅ OK: explicit キャスト + null check + const 指定
void process_matrix(const double* data, int size) {
    if (!data || size <= 0) {
        throw std::invalid_argument("data pointer is null or size is invalid");
    }
    std::vector<int> buffer;
    buffer.reserve(size);
    for (int i = 0; i < size; ++i) {
        if (data[i] > INT_MAX || data[i] < INT_MIN) {
            throw std::overflow_error("value overflow in conversion");
        }
        buffer.push_back(static_cast<int>(data[i]));
    }
}
```

#### メモリ管理
- [ ] 動的メモリが確実に解放されているか（RAII パターン使用）
- [ ] スマートポインタ（`std::unique_ptr` / `std::shared_ptr`）が使用されているか
- [ ] 生ポインタの使用が最小限か

**チェックコード例**:
```cpp
// ❌ NG: 生ポインタ、メモリリーク
class MatrixBuffer {
private:
    double* data;
    int size;
public:
    MatrixBuffer(int n) : size(n) {
        data = new double[n];  // 動的メモリ確保
    }
    // デストラクタが呼ばれなかった場合、メモリリーク
    ~MatrixBuffer() {
        // delete[] data; ← 忘れた！
    }
};

// ✅ OK: スマートポインタ（RAII パターン）
class MatrixBuffer {
private:
    std::unique_ptr<double[]> data;
    int size;
public:
    MatrixBuffer(int n) : size(n), data(std::make_unique<double[]>(n)) {
        // スマートポインタが自動で管理
    }
    // デストラクタで自動的に delete[] が呼ばれる
    ~MatrixBuffer() = default;
};
```

#### STL 利用
- [ ] 適切なコンテナが選択されているか（vector / map / unordered_map）
- [ ] イテレータの有効性が保証されているか
- [ ] range-based for ループが使用されているか

**チェックコード例**:
```cpp
// ❌ NG: コンテナの再確保によるイテレータ無効化
std::vector<int> values;
for (auto it = values.begin(); it != values.end(); ++it) {
    if (*it > 100) {
        values.push_back(-1);  // push_back がメモリ再確保し、イテレータが無効に！
    }
}

// ✅ OK: 事前に容量予約 + range-based for ループ
std::vector<int> values = {10, 50, 150, 200};
std::vector<int> negatives;
for (int val : values) {  // range-based for (イテレータ無効化なし)
    if (val > 100) {
        negatives.push_back(-1);
    }
}
```

### 9.4 パッケージ依存関係の検証

`requirements.txt` の内容を検証します。

**チェック項目**:
- [ ] 使用しているパッケージがすべて `requirements.txt` に listed されているか
- [ ] pinned version (`==`) が使用されているか（または明確な理由で flexible range）
- [ ] 子依存（transitive dependency）が明記されているか
- [ ] セキュリティアップデートが適用されているか

**依存関係の自動検出**:
```python
# scripts/check_requirements.py
import ast
import sys
from pathlib import Path

def extract_imports(py_files):
    """Python ファイルから import を抽出。"""
    imports = set()
    for py_file in py_files:
        with open(py_file) as f:
            try:
                tree = ast.parse(f.read())
                for node in ast.walk(tree):
                    if isinstance(node, ast.Import):
                        for alias in node.names:
                            imports.add(alias.name.split('.')[0])
                    elif isinstance(node, ast.ImportFrom):
                        if node.module:
                            imports.add(node.module.split('.')[0])
            except SyntaxError:
                continue
    return imports

def load_requirements():
    """requirements.txt からパッケージ名を抽出。"""
    with open("docker/requirements.txt") as f:
        requirements = set()
        for line in f:
            line = line.strip()
            if line and not line.startswith("#"):
                # "package==version" から "package" を抽出
                pkg_name = line.split("==")[0].split(">=")[0].split("<")[0]
                requirements.add(pkg_name.lower().replace("-", "_"))
    return requirements

# 使用例
py_files = list(Path("python").rglob("*.py"))
imported = extract_imports(py_files)
required = load_requirements()

missing = imported - required
if missing:
    print(f"⚠️  Missing in requirements.txt: {missing}")
else:
    print("✅ All imports are in requirements.txt")
```

## 7. 実装 ⟷ ドキュメント ⟷ テストの三点セット検証（全体的）

このセクションは、個別ファイルの品質レビューを超えて、機能単位（function / class / module）で **実装・ドキュメント・テストが一貫性を持つか** を検証します.。

### 6.1 機能単位の完全性チェック

**各 public API（関数・クラス）について以下を確認**：

| 観点 | 実装 | ドキュメント | テスト | チェック項目 |
|------|------|------|------|------|
| **インターフェース定義** | ✅ 関数シグネチャ | ✅ Args / Returns | ✅ 呼び出し方 | シグネチャとドキュメントが一貫しているか |
| **前提条件（Precondition）** | ✅ 入力チェック | ✅ docstring | ✅ テスト | 前提条件に違反した入力が reject されるか |
| **戻り値（Postcondition）** | ✅ return 文 | ✅ Returns セクション | ✅ 戻り値の assert | 戻り値の仕様がドキュメントと一致しているか |
| **エラー処理** | ✅ raise 文 | ✅ Raises セクション | ✅ except ブロック | 例外仕様が完全にテストされているか |
| **副作用（Side Effects）** | ✅ 状態変更 | ✅ 記述あるか | ✅ テスト | 副作用がドキュメント化されているか |

### 6.2 ドキュメント ⟷ テストの対応確認

**Docstring の各セクションがテストでカバーされているか**:

```python
# 実装

    """行列逆行列を計算。
    
    Args:
        A: n×n 正則行列     ← テスト: 正則行列でテスト必須
        method: 計算方法。"lu" または "svd" (デフォルト: "lu")
                           ← テスト: 両方法の分岐をテスト必須
                           
    Returns:
        A の逆行列              ← テスト: A @ result ≈ I を assert 必須
        
    Raises:
        ValueError: A が正方でない、またはシンギュラーな場合
                    ← テスト: ValueError を raise すること確認必須
        NotImplementedError: method が未対応の場合
                    ← テスト: NotImplementedError をキャッチ必須
    """
    if A.shape[0] != A.shape[1]:
        raise ValueError("A must be square")
    if A.shape[0] == 0:
        raise ValueError("A must be non-empty")
    
    if method == "lu":
        return _inverse_lu(A)      ← テスト: "lu" パスをカバー必須
    elif method == "svd":
        return _inverse_svd(A)     ← テスト: "svd" パスをカバー必須
    else:
        raise NotImplementedError(f"Method '{method}' not implemented")
        ← テスト: 未対応メソッドでテスト必須

# テスト
def test_matrix_inverse_returns_inverse():
    """Returns セクション: 逆行列が正しく計算されることを検証。"""
    A = jnp.array([[1.0, 2.0], [3.0, 4.0]])
    A_inv = matrix_inverse(A)
    identity = A @ A_inv
    assert jnp.allclose(identity, jnp.eye(2), atol=1e-10)

def test_matrix_inverse_lu_vs_svd():
    """Args セクション: 両 method で結果が等しいことを検証。"""
    A = jnp.array([[1.0, 2.0], [3.0, 4.0]])
    A_inv_lu = matrix_inverse(A, method="lu")
    A_inv_svd = matrix_inverse(A, method="svd")
    assert jnp.allclose(A_inv_lu, A_inv_svd, atol=1e-8)

def test_matrix_inverse_raises_for_singular():
    """Raises セクション: シンギュラー行列が ValueError を raise。"""
    A = jnp.array([[1.0, 2.0], [1.0, 2.0]])  # 特異行列
    with pytest.raises(ValueError):
        matrix_inverse(A)

def test_matrix_inverse_raises_for_non_square():
    """Raises セクション: 非正方行列が ValueError を raise。"""
    A = jnp.array([[1.0, 2.0, 3.0]])
    with pytest.raises(ValueError, match="square"):
        matrix_inverse(A)

def test_matrix_inverse_raises_for_unknown_method():
    """Raises セクション: 未対応 method が NotImplementedError を raise。"""
    A = jnp.array([[1.0, 2.0], [3.0, 4.0]])
    with pytest.raises(NotImplementedError, match="unknown"):
        matrix_inverse(A, method="unknown")
```

```python
# scripts/check_doc_test_triplet.py
import ast
import re
import inspect
from typing import Set

def extract_exceptions_from_docstring(func) -> Set[str]:
    """Docstring から Raises セクションの例外を抽出。"""
    docstring = func.__doc__ or ""
    match = re.search(r"Raises:\s*(.*?)(?=\n\n|\Z)", docstring, re.DOTALL)
    if not match:
        return set()
    
    exceptions = re.findall(r"(\w+Error|\w+Exception)", match.group(1))
    return set(exceptions)

def extract_raised_exceptions(func) -> Set[str]:
    """実装から raise されている例外を抽出（AST 解析）。"""
    try:
        source = inspect.getsource(func)
        tree = ast.parse(source)
        exceptions = set()
        for node in ast.walk(tree):
            if isinstance(node, ast.Raise) and node.exc:
                if isinstance(node.exc, ast.Call):
                    if isinstance(node.exc.func, ast.Name):
                        exceptions.add(node.exc.func.id)
        return exceptions
    except:
        return set()

def check_consistency(module):
    """モジュール内の全関数について一貫性チェック。"""
    for name, func in inspect.getmembers(module, inspect.isfunction):
        if name.startswith("_"):
            continue
        
        documented = extract_exceptions_from_docstring(func)
        implemented = extract_raised_exceptions(func)
        
        # docstring に記載の例外が実装されているか
        missing_impl = documented - implemented
        if missing_impl:
            print(f"⚠️  {name}: Docstring に記載が実装にない: {missing_impl}")
        
        # 実装の例外が docstring に記載されているか
        missing_doc = implemented - documented
        if missing_doc:
            print(f"⚠️  {name}: 実装の例外が Docstring に未記載: {missing_doc}")
```
